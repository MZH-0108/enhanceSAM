# GitHub SSH 配置命令 - 按顺序执行

# ========================================
# 步骤 1: 生成 SSH 密钥
# ========================================
ssh-keygen -t ed25519 -C "2865574175@qq.com"
# 提示时直接按 Enter（3次）

# ========================================
# 步骤 2: 复制公钥到剪贴板
# ========================================
cat ~/.ssh/id_ed25519.pub | clip
# 或者手动复制：
cat ~/.ssh/id_ed25519.pub

# ========================================
# 步骤 3: 添加到 GitHub
# ========================================
# 访问: https://github.com/settings/keys
# 点击 "New SSH key"
# Title: enhanceSAM-laptop
# Key: 粘贴公钥
# 点击 "Add SSH key"

# ========================================
# 步骤 4: 测试 SSH 连接
# ========================================
ssh -T git@github.com
# 首次连接输入 yes
# 看到 "Hi MZH-0108!" 表示成功

# ========================================
# 步骤 5: 创建 GitHub 仓库
# ========================================
# 访问: https://github.com/new
# Repository name: enhanceSAM
# 不要勾选任何初始化选项
# 点击 "Create repository"

# ========================================
# 步骤 6: 配置远程仓库并推送
# ========================================
cd E:\SCIopus\enhanceSAM
git remote add origin git@github.com:MZH-0108/enhanceSAM.git
git branch -M main
git push -u origin main

# ========================================
# 步骤 7: 验证
# ========================================
# 访问: https://github.com/MZH-0108/enhanceSAM
# 检查代码是否已上传

# ========================================
# 完成！以后推送只需：
# ========================================
git add .
git commit -m "your message"
git push
